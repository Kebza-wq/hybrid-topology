ipv6 address 2001:db8:4::1/64
interface GigabitEthernet0/0.20
encapsulation dot1Q 20
ip address 192.168.14.1 255.255.255.0
ipv6 address 2001:db8:4:10::1/64

no ip address
no shutdown
interface Serial0/0/0

ipv6 address 2001:db8:10:34::2/64
no shutdown
interface Serial0/0/1

ipv6 address 2001:db8:10:14::2/64
no shutdown
interface Serial0/1/0
ip address 10.0.45.1 255.255.255.252

no shutdown

router ospf 1
network 192.168.4.0 0.0.0.255 area 0

network 10.0.14.0 0.0.0.3 area 0

ipv6 router ospf 1
interface GigabitEthernet0/0.10

interface GigabitEthernet0/0.20
ipv6 ospf 1 area 0
interface Serial0/0/0
ipv6 ospf 1 area 0
interface Serial0/0/1
ipv6 ospf 1 area 0
interface Serial0/1/0
ipv6 ospf 1 area 0
ip dhcp pool SITE4-PC
